Exemplo de uso nos botões principais
tsx
Copiar
Editar
import { motion } from 'framer-motion';

const AnimatedButton = ({ children, className, ...props }) => (
  <motion.button
    whileHover={{ scale: 1.05 }}
    whileTap={{ scale: 0.97 }}
    className={className}
    {...props}
  >
    {children}
  </motion.button>
);

export default AnimatedButton;
